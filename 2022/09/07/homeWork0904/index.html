2、递归需要满足的条件

  1. 存在边界，当满足某个条件时，递归便不再继续
  2. 每次递归之后，数据规模一直在减小

3、递归和分治的区别

递归和分治本身就不是同一种东西，递归是敲代码的技巧之一，分治是算法的思想之一。

4、常见算法复杂度分析及举例

大O表示法（Big O）：算法对于数据规模 n 的复杂度（时间复杂度、空间复杂度）

  • 时间复杂度：估算程序指令的执行次数（执行时间）
  • 空间复杂度：估计所需占用的存储空间（内存空间）

大O 表示法的规则：

  • 忽略常数、系数、低阶
  • 注意：大O 表示法仅仅是一种粗略、近似的模型，是一种估算；

4.1、时间复杂度

时间复杂度的优先级：

O(1) < O(logn) < O(n) < O(nlogn) < O (n2) < O(n3) < O (2^n)

4.2、空间复杂度

空间复杂度：算法执行时占用的内存空间（栈内存）

4.3、冒泡排序

时间复杂度： O(N^2)

public class BubbleSort implements IArraySort {

    @Override
    public int[] sort(int[] sourceArray) throws Exception {
        // 对 arr 进行拷贝，不改变参数内容
        int[] arr = Arrays.copyOf(sourceArray, sourceArray.length);

        for (int i = 1; i < arr.length; i++) {
            // 设定一个标记，若为true，则表示此次循环没有进行交换，也就是待排序列已经有序，排序已经完成。
            boolean flag = true;

            for (int j = 0; j < arr.length - i; j++) {
                if (arr[j] > arr[j + 1]) {
                    int tmp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = tmp;

                    flag = false;
                }
            }

            if (flag) {
                break;
            }
        }
        return arr;
    }
}

4.4、选择排序

时间复杂度： O(N^2) 

public class SelectionSort implements IArraySort {

    @Override
    public int[] sort(int[] sourceArray) throws Exception {
        int[] arr = Arrays.copyOf(sourceArray, sourceArray.length);

        // 总共要经过 N-1 轮比较
        for (int i = 0; i < arr.length - 1; i++) {
            int min = i;

            // 每轮需要比较的次数 N-i
            for (int j = i + 1; j < arr.length; j++) {
                if (arr[j] < arr[min]) {
                    // 记录目前能找到的最小值元素的下标
                    min = j;
                }
            }

            // 将找到的最小值和i位置所在的值进行交换
            if (i != min) {
                int tmp = arr[i];
                arr[i] = arr[min];
                arr[min] = tmp;
            }

        }
        return arr;
    }
}

package com.lagou.bean;

/**
 * 单链表的节点类
 */
public class Node {

    //名字
    private String name;

    //指向下一个节点
    private Node next;

    public Node() {
    }

    public Node(String name, Node next) {
        setName(name);
        setNext(next);
    }


    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public Node getNext() {
        return next;
    }

    public void setNext(Node next) {
        this.next = next;
    }

    @Override
    public String toString() {
        return "Node{" +
                "name='" + name +
                '}';
    }
}

package com.lagou.bean.packBean;

import com.lagou.bean.Node;

public class SignleLinkedList {

    public static void main(String[] args) {

        SignleLinkedList sll = new SignleLinkedList();
        Node node = new Node("node1",null);
        Node node2 = new Node("node2",null);
        sll.addNode(node);
        sll.addNode(node2);
        sll.show();
        sll.delNode("node2");
        sll.modNode("node","node1-mod");
        sll.show();
    }


    //创建头节点
    private Node head = new Node("head-node",null);

    /**
     * 在链表尾部插入节点
     * @param node
     */
    public void addNode(Node node){

        Node temp;
        temp = head;
        //找到链表尾部
        if( temp.getNext() == null ){
            temp.setNext(node);
        }else{
            while( temp.getNext() != null ){

                temp =  temp.getNext();
            }
        }
        //在尾部插入节点
        temp.setNext(node);
    }


    /**
     * 根据传入的值，删除对应name 的节点
     * @param name
     */
    public void delNode(String name) {

        Node temp;
        temp = head;
        while (true) {
            if (temp.getNext().getName().equals(name) ) {
                //条件匹配，找到想要删除的节点
                temp.setNext( temp.getNext().getNext() );
                break;
            } else {
                temp = temp.getNext();
                if (temp.getNext() == null) {
                    System.out.println("没有找到想要删除的节点！！！");
                    break;
                }
            }
        }
    }

    /**
     * 修改节点的name 为 modName
     * @param name 需要修改的节点 的name的值
     */
    public void modNode(String name,String modName){

        Node temp;
        temp = head;
        while (true) {
            if (temp.getNext().getName().equals(name) ) {
                temp.getNext().setName(modName);
                break;
            } else {
                temp = temp.getNext();
                if (temp.getNext() == null) {
                    System.out.println("没有找到想要修改的节点！！！");
                    break;
                }
            }

        }
    }


    /**
     * 遍历链表
     */
    public void show() {
        Node temp;
        temp = head;
        while (true) {
            if (temp.getNext() == null) {
                System.out.println(temp);
                break;
            } else {
                System.out.println(temp);
                temp = temp.getNext();
            }
        }
        System.out.println();
    }


}
